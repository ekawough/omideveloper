async function analyzeCallSentiment(audioStream) {
  // Initialize neural buffer
  const buffer = await OMI.Neural.createBuffer(audioStream);
  
  // Detect micro-tonality shifts
  const sentiment = await buffer.process({
    sensitivity: 0.95,
    detectObjections: true,
    syncToCRM: true
  });

  return sentiment.mapToUI();
}
